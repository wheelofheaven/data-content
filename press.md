## Brand name — how to write it

<div class="press-usage">
  <div class="press-usage__col press-usage__col--do">
    <h3>Use</h3>
    <ul>
      <li><strong>Wheel of Heaven</strong> — the project</li>
      <li><strong>the Elohim</strong> — always plural ("the Elohim <em>are</em>…")</li>
      <li><strong>Raël</strong>, <strong>Raëlian</strong> — keep the diaeresis</li>
      <li><strong>working hypothesis</strong> — when describing the central claim</li>
      <li><strong>WoH</strong> — only after first mention, in body copy</li>
    </ul>
  </div>
  <div class="press-usage__col press-usage__col--dont">
    <h3>Don't use</h3>
    <ul>
      <li><s>WHEEL OF HEAVEN</s> — all-caps reserved for the brand chip</li>
      <li><s>wheelofheaven</s> — running it together drops legibility</li>
      <li><s>The Elohim is…</s> — the noun is grammatically plural</li>
      <li><s>Raelian</s>, <s>Raelism</s> — keep the umlaut</li>
      <li><s>doctrine</s>, <s>theory</s>, <s>belief system</s> — say <em>reading</em></li>
      <li><s>ancient aliens</s>, <s>secret history</s> — distinct framings we don't share</li>
    </ul>
  </div>

  function copyHandler(e) {

    if (!btn) return;
    e.preventDefault();
    const text = btn.dataset.copy || '';

      try { document.execCommand('copy'); done(); } catch (_) {}
